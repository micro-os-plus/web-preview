515 condition_variable::wait (mutex& mutex)
516 {
517#if defined(OS_TRACE_RTOS_CONDVAR)
518 trace::printf ("%s() @%p %s\n", __func__, this, name ());
519#endif
520
521 // Don't call this from interrupt handlers.
522 os_assert_err(!interrupts::in_handler_mode (), EPERM);
523 // Don't call this from critical regions.
524 os_assert_err(!scheduler::locked (), EPERM);
525
526 thread& crt_thread = this_thread::thread ();
527
528 // Prepare a list node pointing to the current thread.
529 // Do not worry for being on stack, it is temporarily linked to the
530 // list and guaranteed to be removed before this function returns.
531 internal::waiting_thread_node node
532 { crt_thread };
533
534 // TODO: validate
535
536 result_t res;
537 res = mutex.unlock ();
538
539 if (res != result::ok)
540 {
541 return res;
542 }
543
544 {
545 // Add this thread to the condition variable waiting list.
546 list_.link (node);
547 node.thread_->waiting_node_ = &node;
548
549 res = mutex.lock ();
550
551 // Remove the thread from the node waiting list,
552 // if not already removed.
553 node.thread_->waiting_node_ = nullptr;
554 node.unlink ();
555 }
556
557 return res;
