2587 template<typename Allocator>
2588 thread_allocated<Allocator>::thread_allocated (
2589 const char* name, func_t function, func_args_t args,
2590 const attributes& attr, const allocator_type& allocator) :
2591 thread
2592 { name }
2593 {
2594#if defined(OS_TRACE_RTOS_THREAD)
2595 trace::printf ("%s @%p %s\n", __func__, this, this->name ());
2596#endif
2597 if (attr.th_stack_address != nullptr
2598 && attr.th_stack_size_bytes > stack::min_size ())
2599 {
2600 internal_construct_ (function, args, attr, nullptr, 0);
2601 }
2602 else
2603 {
2604 allocator_ = &allocator;
2605
2606 if (attr.th_stack_size_bytes > stack::min_size ())
2607 {
2608 allocated_stack_size_elements_ = (attr.th_stack_size_bytes
2609 + sizeof(typename allocator_type::value_type) - 1)
2610 / sizeof(typename allocator_type::value_type);
2611 }
2612 else
2613 {
2614 allocated_stack_size_elements_ = (stack::default_size ()
2615 + sizeof(typename allocator_type::value_type) - 1)
2616 / sizeof(typename allocator_type::value_type);

2620 // uses allocation_element_t, which is usually larger.
2621 allocated_stack_address_ =
2622 reinterpret_cast<stack::element_t*> ((const_cast<allocator_type&> (allocator)).allocate (
2623 allocated_stack_size_elements_));
2624
2625 assert (allocated_stack_address_ != nullptr);
2626
2627 internal_construct_ (
2628 function,
2629 args,
2630 attr,
2631 allocated_stack_address_,
2632 allocated_stack_size_elements_
2633 * sizeof(typename allocator_type::value_type));
2634 }
