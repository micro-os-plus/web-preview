3614osThreadId
3615osThreadCreate (const osThreadDef_t* thread_def, void* args)
3616{
3617 if (interrupts::in_handler_mode ())
3618 {
3619 return nullptr;
3620 }
3621
3622 if (thread_def == nullptr)
3623 {
3624 return nullptr;
3625 }
3626
3627 thread::attributes attr;
3628 attr.th_priority = thread_def->tpriority;
3629 attr.th_stack_size_bytes = thread_def->stacksize;
3630
3631 // Creating thread with invalid priority should fail (validator requirement).
3632 if (thread_def->tpriority >= osPriorityError)
3633 {
3634 return nullptr;
3635 }
3636
3637 // Find a free slot in the tread definitions array.
3638 for (uint32_t i = 0; i < thread_def->instances; ++i)
3639 {
3640 thread* th = (thread*) &thread_def->data[i];
3641 if (th->state () == thread::state::undefined
3642 || th->state () == thread::state::destroyed)
3643 {
3644 if (attr.th_stack_size_bytes > 0)
3645 {
